

This program demonstrates deleting the head and deleting by key.

```cpp
#include <iostream>

using namespace std;

struct Node {
    int data;
    Node* next;
    Node(int val) : data(val), next(nullptr) {}
};

// --- Helper Functions from previous days ---
void append(Node*& head, int data) {
    Node* newNode = new Node(data);
    if (head == nullptr) {
        head = newNode;
        return;
    }
    Node* current = head;
    while (current->next != nullptr) {
        current = current->next;
    }
    current->next = newNode;
}

void printList(Node* head) {
    Node* current = head;
    cout << "List: ";
    while (current != nullptr) {
        cout << current->data << " -> ";
        current = current->next;
    }
    cout << "nullptr" << endl;
}

// --- Deletion Functions ---

/**
 * Deletes the head node of the list.
 */
void pop_front(Node*& head) {
    // 1. Edge Case: Empty list
    if (head == nullptr) {
        return;
    }
    // 2. Store the current head
    Node* nodeToDelete = head;
    // 3. Move head to the next node
    head = head->next;
    // 4. Deallocate the old head
    delete nodeToDelete;
}

/**
 * Deletes the first node found with the given key.
 */
void deleteNodeByKey(Node*& head, int key) {
    // If the list is empty, do nothing
    if (head == nullptr) return;

    // 1. Edge Case: Head node is the target
    if (head->data == key) {
        pop_front(head);
        return;
    }

    // 2. Traverse to find the node and its predecessor
    Node* current = head;
    Node* previous = nullptr;

    while (current != nullptr && current->data != key) {
        previous = current;
        current = current->next;
    }

    // 4. If key was not found, do nothing
    if (current == nullptr) {
        cout << "Key " << key << " not found in the list." << endl;
        return;
    }

    // 5. Bypass the node
    previous->next = current->next;
    // 6. Deallocate the memory
    delete current;
}


int main() {
    Node* head = nullptr;
    append(head, 10);
    append(head, 20);
    append(head, 30);
    append(head, 40);
    printList(head);

    cout << "\nDeleting the head node (10)..." << endl;
    pop_front(head);
    printList(head);

    cout << "\nDeleting node with key 30..." << endl;
    deleteNodeByKey(head, 30);
    printList(head);
    
    cout << "\nDeleting non-existent node with key 99..." << endl;
    deleteNodeByKey(head, 99);
    printList(head);

    cout << "\nDeleting the new head node (20)..." << endl;
    deleteNodeByKey(head, 20);
    printList(head);

    return 0;
}
```
