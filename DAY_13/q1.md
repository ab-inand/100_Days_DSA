

#### \#\# Hard: Implement Merge Sort

This code is split into the two key functions discussed. `mergeSort` handles the recursive splitting, and `merge` handles the actual sorting and combining.

```cpp
#include <iostream>
#include <vector>

using namespace std;

// The "worker" function that merges two sorted subarrays
void merge(vector<int>& arr, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    // Create temporary vectors (auxiliary space)
    vector<int> L(n1), R(n2);

    // Copy data to temp vectors L[] and R[]
    for (int i = 0; i < n1; i++)
        L[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        R[j] = arr[mid + 1 + j];

    // Merge the temp vectors back into arr[left...right]
    int i = 0; // Initial index of first subarray
    int j = 0; // Initial index of second subarray
    int k = left; // Initial index of merged subarray

    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k] = L[i];
            i++;
        } else {
            arr[k] = R[j];
            j++;
        }
        k++;
    }

    // Copy the remaining elements of L[], if there are any
    while (i < n1) {
        arr[k] = L[i];
        i++;
        k++;
    }

    // Copy the remaining elements of R[], if there are any
    while (j < n2) {
        arr[k] = R[j];
        j++;
        k++;
    }
}

// The "manager" function that recursively divides the array
void mergeSort(vector<int>& arr, int left, int right) {
    // Base Case: If the array has 1 or 0 elements, it's already sorted.
    if (left >= right) {
        return;
    }
    
    // Find the middle point to divide the array into two halves
    int mid = left + (right - left) / 2;

    // Call mergeSort for first half
    mergeSort(arr, left, mid);

    // Call mergeSort for second half
    mergeSort(arr, mid + 1, right);

    // Merge the two halves sorted in above calls
    merge(arr, left, mid, right);
}

void printVector(const vector<int>& arr) {
    for (int val : arr) {
        cout << val << " ";
    }
    cout << endl;
}

int main() {
    vector<int> numbers = {12, 11, 13, 5, 6, 7};
    cout << "Given array is: ";
    printVector(numbers);

    mergeSort(numbers, 0, numbers.size() - 1);

    cout << "Sorted array is: ";
    printVector(numbers);
    
    return 0;
}
```
