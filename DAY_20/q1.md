

#### \#\# Medium: Merge `k` Sorted Arrays

```cpp
#include <iostream>
#include <vector>
#include <queue> // For priority_queue

using namespace std;

// Define a type for our min-heap elements for clarity
using HeapElement = pair<int, pair<int, int>>; // {value, {array_index, element_index}}

vector<int> mergeKSortedArrays(const vector<vector<int>>& arrays) {
    vector<int> result;
    // Create a min-heap
    priority_queue<HeapElement, vector<HeapElement>, greater<HeapElement>> minHeap;

    // Push the first element of each array into the heap
    for (int i = 0; i < arrays.size(); ++i) {
        if (!arrays[i].empty()) {
            minHeap.push({arrays[i][0], {i, 0}});
        }
    }

    // Main loop
    while (!minHeap.empty()) {
        // Get the smallest element
        HeapElement top = minHeap.top();
        minHeap.pop();

        int value = top.first;
        int arrayIndex = top.second.first;
        int elementIndex = top.second.second;

        // Add it to the result
        result.push_back(value);

        // If there's a next element in the same array, push it to the heap
        if (elementIndex + 1 < arrays[arrayIndex].size()) {
            minHeap.push({arrays[arrayIndex][elementIndex + 1], {arrayIndex, elementIndex + 1}});
        }
    }

    return result;
}

int main() {
    vector<vector<int>> arrays = {{1, 4, 5}, {1, 3, 4}, {2, 6}};
    vector<int> merged = mergeKSortedArrays(arrays);
    for (int x : merged) {
        cout << x << " "; // Expected: 1 1 2 3 4 4 5 6
    }
    cout << endl;
    return 0;
}
```
