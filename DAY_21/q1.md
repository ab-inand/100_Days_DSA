

#### \#\# Easy: Remove Duplicates from an Array

The `unordered_set` makes this task trivial. We can leverage its property of only storing unique elements.

```cpp
#include <iostream>
#include <vector>
#include <unordered_set>

using namespace std;

// Function to remove duplicates and return a new vector
vector<int> removeDuplicates(const vector<int>& nums) {
    // Insert all elements from the vector into the set.
    // The set will automatically handle duplicates for us.
    unordered_set<int> uniqueElements(nums.begin(), nums.end());

    // Create a new vector from the unique elements in the set.
    vector<int> result(uniqueElements.begin(), uniqueElements.end());

    return result;
}

void printVector(const vector<int>& vec) {
    for (int val : vec) {
        cout << val << " ";
    }
    cout << endl;
}

int main() {
    vector<int> numbersWithDuplicates = {1, 2, 5, 2, 4, 1, 8, 5, 9};
    
    cout << "Original vector: ";
    printVector(numbersWithDuplicates);
    
    vector<int> uniqueNumbers = removeDuplicates(numbersWithDuplicates);
    
    cout << "Vector with duplicates removed: ";
    printVector(uniqueNumbers); // Note: Order is not guaranteed
    
    return 0;
}
```
