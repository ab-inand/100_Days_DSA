#### \#\# Easy: Print Numbers from N to 1 Recursively

```cpp
#include <iostream>

using namespace std;

void printNto1(int n) {
    // Base Case: If n is 0, there's nothing to print, so stop.
    if (n == 0) {
        return;
    }

    // First, print the current number.
    cout << n << " ";

    // Recursive Step: Call the function for the next smaller number.
    printNto1(n - 1);
}

int main() {
    int startNumber = 5;
    cout << "Printing from " << startNumber << " to 1: ";
    printNto1(startNumber);
    cout << endl;
    return 0;
}
```
