#include <iostream>

using namespace std;

// --- Node struct and helper functions from previous days ---
struct Node {
    int data;
    Node* next;
    Node(int val) : data(val), next(nullptr) {}
};

void append(Node*& head, int data) {
    Node* newNode = new Node(data);
    if (head == nullptr) { head = newNode; return; }
    Node* current = head;
    while (current->next != nullptr) { current = current->next; }
    current->next = newNode;
}

void printList(Node* head) {
    Node* current = head;
    cout << "List: ";
    while (current != nullptr) {
        cout << current->data << " -> ";
        current = current->next;
    }
    cout << "nullptr" << endl;
}

// --- Implementation Tasks ---

/**
 * Reverses a linked list in-place.
 * The head pointer is passed by reference so it can be updated.
 */
void reverseList(Node*& head) {
    Node* prev = nullptr;
    Node* curr = head;
    Node* next = nullptr;

    while (curr != nullptr) {
        // Store the next node before we overwrite curr->next
        next = curr->next;
        // Reverse the current node's pointer
        curr->next = prev;
        // Move prev and curr one step forward for the next iteration
        prev = curr;
        curr = next;
    }
    // After the loop, prev is the new head
    head = prev;
}

/**
 * Finds the middle node of a linked list using the slow/fast pointer approach.
 * Returns a pointer to the middle node.
 */
Node* findMiddle(Node* head) {
    if (head == nullptr) {
        return nullptr;
    }

    Node* slow = head;
    Node* fast = head;

    // The loop condition ensures fast doesn't go past the end
    while (fast != nullptr && fast->next != nullptr) {
        slow = slow->next;
        fast = fast->next->next;
    }

    return slow;
}

int main() {
    Node* head = nullptr;
    append(head, 10);
    append(head, 20);
    append(head, 30);
    append(head, 40);
    append(head, 50);
    printList(head);

    // --- Task 1: Find the middle element ---
    Node* middle = findMiddle(head);
    if (middle != nullptr) {
        cout << "The middle element is: " << middle->data << endl;
    }

    // --- Task 2: Reverse the list ---
    cout << "\nReversing the list..." << endl;
    reverseList(head);
    printList(head);
    
    return 0;
}
