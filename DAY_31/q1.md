

#### \#\# Easy: Simulate a Ticketing System

This program simulates a simple queue for a ticket counter, serving customers in the exact order they arrive (FIFO).

```cpp
#include <iostream>
#include <queue>
#include <string>

using namespace std;

int main() {
    // Create a queue to hold the names of customers
    queue<string> ticket_queue;

    cout << "Ticketing system is now open." << endl;

    // Customers arrive and are added to the back of the queue
    ticket_queue.push("Alice");
    cout << "Alice has joined the queue. Queue size is now: " << ticket_queue.size() << endl;
    
    ticket_queue.push("Bob");
    cout << "Bob has joined the queue. Queue size is now: " << ticket_queue.size() << endl;

    ticket_queue.push("Charlie");
    cout << "Charlie has joined the queue. Queue size is now: " << ticket_queue.size() << endl;

    cout << "\n--- Serving customers ---" << endl;

    // Serve customers as long as the queue is not empty
    while (!ticket_queue.empty()) {
        // 1. Get the customer at the front of the line
        string current_customer = ticket_queue.front();

        // 2. "Serve" the customer
        cout << "Now serving: " << current_customer << endl;

        // 3. Remove the customer from the front of the line
        ticket_queue.pop();

        cout << "Customers remaining in queue: " << ticket_queue.size() << endl;
    }

    cout << "\nTicketing system is now closed. No customers left." << endl;

    return 0;
}
```
